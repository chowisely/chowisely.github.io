#### [문제 바로가기](https://www.acmicpc.net/problem/4386)

#### 접근


```cpp
#include <bits/stdc++.h>
using namespace std;

struct Edge {
  int n1;
  int n2;
  double w;

  Edge(int n1, int n2, double w): n1(n1), n2(n2), w(w) {}
};

struct comp_w {
  bool operator() (Edge const& n1, Edge const& n2) {
    return n1.w > n2.w;
  }
};

struct Star {
  double x;
  double y;
  int idx;

  Star(double x, double y, int i): x(x), y(y), idx(i){}
};


vector<int> parent;


double dist(const Star& a, const Star& b) {
  return sqrt(pow(a.x - b.x, 2) + pow(a.y - b.y, 2));
}

int find(int n) {
  if(parent[n] == n) return n;
  return parent[n] = find(parent[n]);
}

void merge(int n1, int n2) {
  n1 = find(n1);
  n2 = find(n2);
  if(n1 != n2) parent[n2] = n1;
}

bool isCyclic(int n1, int n2) {
  if(find(n1) == find(n2)) return true;
  merge(n1, n2);
  return false;
}


int main() {
  std::ios::sync_with_stdio(false);
  cin.tie(NULL); cout.tie(NULL);
  int n;
  double sx, sy;
  int cnt = 0;
  double cost = 0;
  vector<Star> stars;
  priority_queue<Edge, vector<Edge>, comp_w> pq;

  cin >> n;
  parent.assign(n + 1, 0);
  for(int i = 1; i <= n; i++) {
    cin >> sx >> sy;
    stars.push_back(Star(sx, sy, i));
    parent[i] = i;
  }

  for(int i = 0; i < n - 1; i++)
    for(int j = i + 1; j < n; j++)
      pq.push(Edge(stars[i].idx, stars[j].idx, dist(stars[i], stars[j])));

  while(!pq.empty()) {
    if(cnt == n - 1)
      break;

    Edge e = pq.top();
    pq.pop();

    if(!isCyclic(e.n1, e.n2)) {
      cost += e.w;
      cnt++;
    }
  }

  cout << cost;
}
```
